## Description
A button for forms and other purposes. Provides a wrapper around the native HTMLButtonElement, with
custom styles and additional features including validation.

	"primary",
	"primaryVariant",
	"secondary",
	"secondaryVariant",

## StandardStory
export const STANDARD_SPACING = 8;

<Preview>
	<Story name="standard">
		{() => {
			const onClick = useCallback(action("onClick"), []);
			return (
				<Paper>
					{STANDARD_COLORS.map((color) => (
						<div key={color} style={{ margin: STANDARD_SPACING }}>
							Color: {color.toUpperCase()}
							<br />
							<br />
							{STANDARD_SIZES.map((size) => (
								<span key={`${color}_${size}`} style={{ margin: STANDARD_SPACING }}>
									<Button color={color} size={size} onClick={onClick}>
										{size.toUpperCase()}
									</Button>
								</span>
							))}
						</div>
					))}
				</Paper>
			);
		}}
	</Story>
