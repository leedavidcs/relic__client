and additional features including validation.

## StandardStory
export const STANDARD_MAX_INPUT_LENGTH = 5;

<Preview>
	<Story name="standard">
		{() => {
			const [value0, setValue0] = useState("");
			const [value1, setValue1] = useState("");
			const [value2, setValue2] = useState("");
			const [value3, setValue3] = useState("");
			const validator = useCallback((value) => {
				const isValid = value.length < STANDARD_MAX_INPUT_LENGTH;
				return isValid ? null : "Text is too long";
			}, []);
			const style = { marginBottom: 8 };
			const onChange0 = useCallback(onInputValueChanged(setValue0), []);
			const onChange1 = useCallback(onInputValueChanged(setValue1), []);
			const onChange2 = useCallback(onInputValueChanged(setValue2), []);
			const onChange3 = useCallback(onInputValueChanged(setValue3), []);
			return (
				<Paper>
					<TextInput
						label={"Must be fewer than 5 chars"}
						validator={validator}
						style={style}
						onChange={onChange0}
						value={value0}
					/>
					<TextInput
						label={"Must be fewer than 5 chars"}
						validator={validator}
						style={style}
						onChange={onChange1}
						value={value1}
						variant="outlined"
					/>
					<TextInput
						label={"Must be fewer than 5 chars"}
						validator={validator}
						style={style}
						onChange={onChange2}
						value={value2}
						startIcon={<FaBacon />}
					/>
					<TextInput
						label={"Must be fewer than 5 chars"}
						validator={validator}
						style={style}
						onChange={onChange3}
						value={value3}
						variant="outlined"
						startIcon={<FaBacon />}
					/>
				</Paper>
			);
		}}
	</Story>
</Preview>
